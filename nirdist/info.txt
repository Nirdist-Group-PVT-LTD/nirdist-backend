CREATE TABLE variant (

    v_id INT AUTO_INCREMENT PRIMARY KEY,

    v_name VARCHAR(100) NOT NULL,

    v_birth DATE,

    v_username VARCHAR(50) UNIQUE NOT NULL,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP

);



-- Multi‑value contacts (allow multiple per user)

CREATE TABLE variant_username (

    vu_id INT AUTO_INCREMENT PRIMARY KEY,

    v_id INT NOT NULL,

    username VARCHAR(50) NOT NULL,

    used_from TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE

);



CREATE TABLE variant_email (

    ve_id INT AUTO_INCREMENT PRIMARY KEY,

    v_id INT NOT NULL,

    email VARCHAR(255) NOT NULL,

    is_verified BOOLEAN DEFAULT FALSE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE

);



CREATE TABLE variant_number (

    vn_id INT AUTO_INCREMENT PRIMARY KEY,

    v_id INT NOT NULL,

    phone_number VARCHAR(20) NOT NULL,

    type ENUM('mobile', 'home', 'work') DEFAULT 'mobile',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE

);



-- Follow system (social requirement)

CREATE TABLE follows (

    follow_id INT AUTO_INCREMENT PRIMARY KEY,

    follower_id INT NOT NULL,

    followee_id INT NOT NULL,

    follow_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (follower_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    FOREIGN KEY (followee_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    UNIQUE KEY unique_follow (follower_id, followee_id)

);

-- 2. SOUND & ARTIST (music/sound features)

-- ------------------------------------------------------------

CREATE TABLE sound_artist (

    s_a_id INT AUTO_INCREMENT PRIMARY KEY,

    s_a_name VARCHAR(100) NOT NULL,

    s_a_date DATE

);



CREATE TABLE sound (

    s_id INT AUTO_INCREMENT PRIMARY KEY,

    lyrics TEXT,                          -- .lrc or plain text

    s_link VARCHAR(500) NOT NULL,

    s_a_id INT,                           -- artist FK

    s_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (s_a_id) REFERENCES sound_artist(s_a_id) ON DELETE SET NULL

);



-- User‑produced sounds (original audio)

CREATE TABLE sound_produce (

    sp_id INT AUTO_INCREMENT PRIMARY KEY,

    sp_name VARCHAR(100) NOT NULL,

    sp_discription TEXT NOT NULL,

    sp_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    v_id INT NOT NULL,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE

);



-- ------------------------------------------------------------

-- 3. POSTS (core content)

-- ------------------------------------------------------------

CREATE TABLE post (

    p_id INT AUTO_INCREMENT PRIMARY KEY,

    v_id INT NOT NULL,

    post_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    statuse BOOLEAN DEFAULT TRUE,          -- published/hidden

    reach INT DEFAULT 0,

    a_id VARCHAR(50),                      -- algorithm ID (optional)

    discription TEXT NOT NULL,

    s_id INT NULL,                         -- optional attached sound

    p_update TIMESTAMP NULL ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    FOREIGN KEY (s_id) REFERENCES sound(s_id) ON DELETE SET NULL,

    INDEX idx_post_time (post_time),

    INDEX idx_v_id (v_id)

);



-- Post media (multiple images/videos per post)

CREATE TABLE post_media (

    p_m_id INT AUTO_INCREMENT PRIMARY KEY,

    p_id INT NOT NULL,

    p_m_link VARCHAR(500) NOT NULL,

    p_m_upload_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    p_m_display BOOLEAN DEFAULT TRUE,      -- currently in use

    FOREIGN KEY (p_id) REFERENCES post(p_id) ON DELETE CASCADE,

    INDEX idx_p_id (p_id)

);



-- Reactions on posts (like, love, etc.)

CREATE TABLE post_reaction (

    pr_id INT AUTO_INCREMENT PRIMARY KEY,

    p_id INT NOT NULL,

    v_id INT NOT NULL,

    pr_type ENUM('like', 'love', 'haha', 'wow', 'sad', 'angry') NOT NULL,

    pr_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (p_id) REFERENCES post(p_id) ON DELETE CASCADE,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    UNIQUE KEY unique_post_reaction (p_id, v_id),

    INDEX idx_pr_type (pr_type)

);



-- Comments on posts

CREATE TABLE post_comment (

    post_comment_id INT AUTO_INCREMENT PRIMARY KEY,

    post_id INT NOT NULL,

    v_id INT NOT NULL,

    post_comment_text TEXT NOT NULL,

    post_comment_media VARCHAR(500),       -- optional image/video in comment

    parent_comment_id INT NULL,            -- for nested replies

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (post_id) REFERENCES post(p_id) ON DELETE CASCADE,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    FOREIGN KEY (parent_comment_id) REFERENCES post_comment(post_comment_id) ON DELETE CASCADE,

    INDEX idx_post_id (post_id)

);



-- Reactions on comments

CREATE TABLE post_comment_reaction (

    pcr_id INT AUTO_INCREMENT PRIMARY KEY,

    comment_id INT NOT NULL,

    v_id INT NOT NULL,

    reaction_type ENUM('like', 'love', 'haha', 'wow', 'sad', 'angry') NOT NULL,

    reaction_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (comment_id) REFERENCES post_comment(post_comment_id) ON DELETE CASCADE,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    UNIQUE KEY unique_comment_reaction (comment_id, v_id)

);



-- Shares

CREATE TABLE post_share (

    p_s_id INT AUTO_INCREMENT PRIMARY KEY,

    p_id INT NOT NULL,

    v_id INT NOT NULL,

    p_s_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    p_s_to INT NULL,                       -- NULL = public, else user ID (variant)

    FOREIGN KEY (p_id) REFERENCES post(p_id) ON DELETE CASCADE,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    FOREIGN KEY (p_s_to) REFERENCES variant(v_id) ON DELETE SET NULL,

    INDEX idx_share_time (p_s_time)

);



-- Saves (bookmarks)

CREATE TABLE post_save (

    p_sa_id INT AUTO_INCREMENT PRIMARY KEY,

    p_id INT NOT NULL,

    v_id INT NOT NULL,

    p_sa_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    p_sa_type ENUM('bookmark', 'watch_later') DEFAULT 'bookmark',

    FOREIGN KEY (p_id) REFERENCES post(p_id) ON DELETE CASCADE,

    FOREIGN KEY (v_id) REFERENCES variant(v_id) ON DELETE CASCADE,

    UNIQUE KEY unique_save (p_id, v_id)

);

-- 4. STORIES (ephemeral content)

CREATE TABLE story (

    s_id INT AUTO_INCREMENT PRIMARY KEY,

Based on the database schema you provided, this application is a media-heavy social networking platform with strong similarities to Instagram or TikTok. It revolves around multimedia posts, 24-hour ephemeral stories, short text notes, a robust audio/sound system, and highly interactive engagement (reactions, comments, shares, saves).

Here is a comprehensive UI/UX architecture and wireframe layout tailored for a mobile phone experience to support all these database features.

1. Global Navigation (Bottom Tab Bar)
To keep the app accessible and intuitive, the main navigation should sit at the bottom of the screen.

🏠 Home: Main feed (Posts, Stories, Notes).

🔍 Discover: Search for users, trending sounds, and algorithms.

➕ Create: Central hub for posting media, stories, notes, or recording original sounds.

❤️ Activity: Notifications for reactions, comments, follows, and shares.

👤 Profile: User dashboard, settings, and personal archives.

2. Screen-by-Screen UI Layout
🏠 Home Screen (The Core Feed)
Top Bar (Horizontal Scroll):

Your Note (+): A floating text bubble over the user's avatar to add a quick thought (notes table).

Friends' Notes: Text bubbles over mutuals' avatars.

Stories Banner: Circular avatars surrounded by a gradient ring indicating active 24-hour stories (story table).

Main Feed (Vertical Scroll):

Each Post takes up most of the screen (or acts as a card).

Header: User Avatar, Username (variant_username), and post time (post_time).

Media Container: Swipeable carousel for multiple images or a full-screen video player (post_media).

Footer Details: Caption/Description (discription).

Audio Marquee: A scrolling text bar at the bottom showing the attached track or user-produced sound (sound or sound_produce), tapping it goes to the Audio Page.

Floating Action Column (Right side or Bottom row):

React: A heart icon. UX Interaction: Tap to "Like", long-press to reveal the reaction menu: Love, Haha, Wow, Sad, Angry (post_reaction).

Comment: Opens a bottom sheet showing nested replies (post_comment).

Share: Arrow icon to share publicly or directly to a specific user (post_share).

Save: Bookmark icon. Long-press to choose "Bookmark" or "Watch Later" (post_save).

🎵 Audio / Sound Page (Triggered by tapping a Post's Sound)
Header: Album art or User Avatar.

Track Info: Song Name & Artist (sound_artist) OR Custom Sound Name & Description (sound_produce).

Action Button: "Use this Sound" (Jumps to Create camera).

Grid: A 3-column masonry grid of all post_media thumbnails that use this s_id.

➕ Create Flow (The Studio)
When the user taps the center + button, a bottom sheet or full-screen camera opens with a slider at the bottom:

Post: Upload/record media, write a description, attach a sound (s_id), and publish (statuse = TRUE).

Story: Capture photo/video. Auto-sets expiration to 24 hours (expires_at).

Note: Simple text input interface with visibility toggles (Public, Friends, Only Me) (visibility).

Sound: Audio recorder interface to create a custom track with a title and description (sound_produce).

💬 Comment Section (Bottom Sheet Overlay)
Input Field: Text area with an attachment icon for media (post_comment_media).

Comment Thread:

Displays parent comments.

"View Replies" button to expand nested children (parent_comment_id).

Small reaction icons next to comments for Like, Love, Haha, etc. (post_comment_reaction).

👤 Profile Screen (The User Hub)
Header Profile:

Avatar, Display Name (v_name), and current active Username (username).

Stats row: Following, Followers (follows table), Total Reach (reach).

Action Buttons: Edit Profile, Share Profile, Follow/Unfollow.

Content Tabs:

Grid (Posts): 3-column view of active posts.

Sounds (Originals): A list of audio tracks this user has created (sound_produce).

Saved (Private): Organized by "Bookmarks" and "Watch Later" (p_sa_type).

3. Key UX Interactions & Logic
Algorithmic Sorting (a_id): The Home feed should have a toggle at the very top (e.g., "Following" vs. "For You"). The "For You" feed utilizes the a_id from the post table to serve content based on reach and engagement.

Data Deletion Cascades: Since your schema heavily utilizes ON DELETE CASCADE, the UI needs robust confirmation dialogues (e.g., "Are you sure you want to delete this post? All media, comments, and reactions will be permanently removed.").

Identity Management: In the "Edit Profile" settings, provide a UI to manage multi-value contacts.

A list view showing all past/present Usernames (variant_username).

Verified badges next to Emails (is_verified in variant_email).

Icons designating phone numbers as Mobile, Home, or Work (type in variant_number).

Story Expiration: Stories require a progress bar at the top of the screen that depletes as the media plays. Once expires_at is reached, the UI automatically hides the content from the global feed.

4. Visual Design Recommendations
Dark Mode First: Because this schema points to a media-heavy application (images, videos, stories), a dark background (Deep Grays or True Black) will make the post_media pop and reduce eye strain.

Color Coding Reactions: Give the diverse reactions distinct visual identities to make the activity feed skimmable:

❤️ Love (Red)

😂 Haha (Yellow)

😲 Wow (Purple)

😢 Sad (Blue)

😡 Angry (Orange/Red)

Typography: Use a clean, sans-serif font (like Inter, Roboto, or San Francisco) to ensure high legibility for densely packed text like notes and post_comment.

To translate your database schema into a high-fidelity mobile experience, the UI is divided into five core layers. This "full UI" focuses on Media-First design, ensuring that sounds, stories, and posts interact seamlessly.

1. The Home Feed (The "Variant" Stream)
The home screen serves as the primary consumption hub, pulling from post, story, and notes.

Top Bar (Notes & Stories): * Text Bubbles: Small, circular text bubbles above user avatars showing content from the notes table.

Story Rings: Traditional circular avatars with colored borders. Tapping opens the full-screen story view.

The Post Card: * Media Carousel: Since post_media allows multiple links, use a horizontal swipe indicator (dots) at the bottom of the image.

Sound Marquee: A scrolling "music note" bar at the bottom of the post showing s_a_name (Artist) or sp_name (User-produced sound).

Dynamic Reactions: A single tap on the "Heart" logs a like. A long-press triggers a haptic pop-up showing the pr_type ENUM options (Love, Haha, Wow, etc.).

2. Interactive Discovery (The "Search & Algorithm" Hub)
This utilizes the a_id (Algorithm ID) and reach fields to categorize content.

Grid View: A 3-column masonry grid of images and videos.

Audio Search: A dedicated tab for sound and sound_artist. Users can preview lyrics (lyrics) and see how many post entries are using a specific s_id.

Trending Notes: A ticker-style feed of public notes that have high notes_reaction counts.

3. The "Create" Studio (Central Action)
Triggered by the center + button, this interface handles all data entry.

Mode Selector: A sliding bottom menu to switch between:

Post: Uploading media to post_media and selecting a sound.

Story: Camera interface with a 24-hour countdown preview (visualizing expires_at).

Produce Sound: A waveform recorder for the sound_produce table, allowing users to name and describe their original audio.

Note: A minimalist text input with a "Visibility" toggle (Mapping to the visibility ENUM: Public, Friends, Only Me).

4. Engagement Layers (Comments & Shares)
These are handled via Bottom Sheets to keep the user in the context of the post.

Threaded Comments: Displays post_comment_text. If parent_comment_id is present, the reply is slightly indented with a "Thread" line.

Share Menu: Shows a list of recent follows. Tapping a user creates a post_share entry with p_s_to set to that user’s ID.

Saves: A bookmark icon with two folders: "General Bookmarks" and "Watch Later" (Mapping to p_sa_type).

5. User Profile (The Identity Dashboard)
This screen manages the complex variant and variant_username data.

Section	UI Element	Database Link
Identity	Display Name & Current Username	v_name, v_username
History	"Past Usernames" dropdown	variant_username table
Metrics	Follower/Following Count	follows table
Contact	Verified Email icon / Phone badges	is_verified, variant_number.type
Grid	Personal Posts & Produced Sounds	post & sound_produce
UX Design Principles Applied:
Ephemeral Awareness: Stories that are close to expires_at (e.g., within 1 hour) could have a glowing red border to create urgency.

Audio Integration: The app includes a global "Mini-Player" at the bottom if a user navigates away from a post while a sound is playing.

Clean Architecture: Despite the complex multi-table contact system (variant_email, variant_number), the UI hides these in a "Contact Info" drawer to keep the profile sleek.
